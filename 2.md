@echo off
rem 设置ffmpeg.exe的路径
set ffmpeg_path=C:\ffmpeg\bin\ffmpeg.exe
rem 设置输出文件夹的路径
set output_path=C:\output
rem 遍历F:\videos目录及其子目录下的所有mp4文件
for /R "F:\videos" %%i in (*.mp4) do (
    rem 获取文件名（不含扩展名）
    set file_name=%%~ni
    rem 调用ffmpeg.exe将mp4文件转换为avi文件
    %ffmpeg_path% -i "%%i" "%output_path%\!file_name!.avi"
)
